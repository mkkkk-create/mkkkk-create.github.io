def dec(enc):
    Alphabat = "0123456789abcdef"
    reverse_map = {}
    for i in range(len(Alphabat)):
        new_index = ((i * 5) + 3) % 16
        reverse_map[Alphabat[new_index]] = Alphabat[i]

    dec_str = []
    for char in enc:
        if char in reverse_map:
            dec_str.append(reverse_map[char])
        else:
            dec_str.append(char)

    return ''.join(dec_str)
