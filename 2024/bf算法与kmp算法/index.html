int INDEX_BF(string s, string t, int pos){
    i = pos;
    j = 1;
    while (i <= s.length() && j <= t.length()){
        if (s[i] == t[j]){
            i++;
            j++;
        }
        else{
            i = i - j + 2;
            j = 1;
        }
    }
    if (j > t.length()) return i - t.length();
    else 
        return 0;
}
